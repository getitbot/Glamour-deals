        function handleCopyClick(event) {
            const button = event.currentTarget;
            const coupon = button.dataset.coupon;

            // Find the "Buy" link. It is the next sibling of the buttons' container div.

                buyLink.classList.add('visible');

                showToast();
                button.textContent = 'Copied!';

        }

        function showToast() {
